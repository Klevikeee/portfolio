Biztonsági záras LED vezérlő (Keypad + I2C LCD)

Feladat leírása:

  • Egy 4x4-es mátrix billentyűzet (Keypad) és egy 16x2-es I2C LCD kijelző segítségével egy PIN-kódos beléptető rendszer szimulálása.
  • A rendszer egy 4 számjegyű kódot kér a felhasználótól. Az alapértelmezett PIN kód (6767), amelyet az Arduino belső EEPROM memóriájában tárol, így áramszünet (vagy újraindítás) esetén sem felejti el a beállított jelszót.
  • Sikeres belépés után a felhasználó a billentyűzet gombjaival (1-8) különböző LED-eket vagy LED-csoportokat kapcsolhat fel, a '0' gombbal pedig lekapcsolhatja az összeset.
  • A '*' gomb megnyomásával bármikor biztonságosan ki lehet jelentkezni a rendszerből (a LED-ek ilyenkor lekapcsolnak).
  • A 'C' gomb megnyomásával a bejelentkezett felhasználó megváltoztathatja a PIN kódot, amit a rendszer azonnal elment az EEPROM-ba.

Megvalósítás:

A projekt három fő külső könyvtárra épül: a Keypad.h a hardveres gombbeolvasásért felel, a LiquidCrystal_I2C.h a kijelző mindössze két adatvezetéken (SDA, SCL) történő vezérlését végzi, az EEPROM.h pedig a jelszó tárolását biztosítja.

A kód gerincét egy állapotgép (State Machine) adja a loop() függvényben, amely a belepve logikai változó alapján dönti el, hogy a felhasználótól a jelszót várja (hitelesítési fázis), vagy már a vezérlőparancsokat fogadja. A LED-ek a D2-D5 és a D10-D13 digitális lábakra lettek bekötve, a Keypad pedig analóg (A0-A3 sorok) és digitális (D6-D9 oszlopok) lábakat használ.

A vezérlő forráskódja:

#include <Keypad.h>
#include <LiquidCrystal_I2C.h>
#include <Wire.h>
#include <EEPROM.h>

LiquidCrystal_I2C lcd(0x27, 16, 2);
const byte ROWS = 4;
const byte COLS = 4;

char keys[ROWS][COLS] = {
  {'1','2','3','A'},
  {'4','5','6','B'},
  {'7','8','9','C'},
  {'*','0','#','D'}
};

byte rowPins[ROWS] = {A0, A1, A2, A3};
byte colPins[COLS] = {6, 7, 8, 9};

Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

String helyeskod = "";
String beirtkod = "";
String ujkod = "";
bool belepve = false;
bool kodmodositas = false;

void setup() {
  lcd.init();
  lcd.backlight();
  Serial.begin(9600);

  // EEPROM kód beolvasása, ha nem szám, alapértelmezett "6767" beállítása
  for (int i = 0; i < 4; i++) {
    char c = EEPROM.read(i);
    if (c >= '0' && c <= '9') {
      helyeskod += c;
    } else {
      helyeskod = "6767";
      break;
    }
  }

  // LED lábak kimenetté alakítása
  for (int i = 2; i = 5; i++) pinMode(i, OUTPUT);
  for (int i = 10; i <= 13; i++) pinMode(i, OUTPUT);
}

void loop() { 
  if (!belepve) {
    char kod = keypad.getKey();
    lcd.setCursor(0, 0);
    lcd.print("Add meg a kodot:");

    if (kod && kod >= '0' && kod <= '9') {
      if (beirtkod.length() < 4) {
        beirtkod += kod;
        lcd.setCursor(5, 1);
        lcd.print(beirtkod);
      }

      if (beirtkod.length() == 4) {
        if (beirtkod == helyeskod) {
          lcd.clear();
          lcd.setCursor(5, 1);
          lcd.print("OK");
          delay(1500);
          lcd.clear();
          lcd.print("Valassz LED-et");
          belepve = true;
        } else {
          lcd.clear();
          lcd.setCursor(4, 1);
          lcd.print("Rossz kod");
          delay(1500);
          lcd.clear();
        }
        beirtkod = "";
      }
    }
  }
  else {
    char kod = keypad.getKey();

    // Kijelentkezés
    if (kod == '*') {
      belepve = false;
      beirtkod = "";
      ujkod = "";
      kodmodositas = false;
      lcd.clear();
      lcd.print("Kijelentkezve");
      for (int i = 2; i <= 5; i++) digitalWrite(i, LOW);
      for (int i = 10; i <= 13; i++) digitalWrite(i, LOW);
      delay(1500);
      lcd.clear();
      return;
    }

    // Jelszó módosítása
    if (kod == 'C') {
      kodmodositas = true;
      ujkod = "";
      lcd.clear();
      lcd.print("Uj PIN:");
      return;
    }

    if (kodmodositas) {
      if (kod >= '0' && kod <= '9') {
        if (ujkod.length() < 4) {
          ujkod += kod;
          lcd.setCursor(0, 1);
          lcd.print(ujkod);
        }

        if (ujkod.length() == 4) {
          helyeskod = ujkod;
          for (int i = 0; i < 4; i++) {
            EEPROM.write(i, ujkod[i]);
          }
          kodmodositas = false;
          lcd.clear();
          lcd.print("PIN mentve");
          delay(1500);
          lcd.clear();
          lcd.print("Valassz LED-et");
        }
      }
      return;
    }

    // LED-ek vezérlése (1-8, 0)
    if (kod >= '0' && kod <= '9') {
      if (kod == '1') {
        digitalWrite(2, HIGH);
        for (int i = 3; i <= 13; i++) digitalWrite(i, LOW);
        lcd.setCursor(0,1);
        lcd.print("Felkapcsolva :");
        lcd.print(kod);
      } else if (kod == '2') {
        digitalWrite(2, HIGH); digitalWrite(3, HIGH);
        for (int i = 4; i <= 13; i++) digitalWrite(i, LOW);
        lcd.setCursor(0,1);
        lcd.print("Felkapcsolva :");
        lcd.print(kod);
      } else if (kod == '3') {
        digitalWrite(2, HIGH); digitalWrite(3, HIGH); digitalWrite(4, HIGH);
        for (int i = 5; i <= 13; i++) digitalWrite(i, LOW);
        lcd.setCursor(0,1);
        lcd.print("Felkapcsolva :");
        lcd.print(kod);
      } else if (kod == '4') {
        for (int i = 2; i <= 5; i++) digitalWrite(i, HIGH);
        for (int i = 6; i <= 13; i++) digitalWrite(i, LOW);
        lcd.setCursor(0,1);
        lcd.print("Felkapcsolva :");
        lcd.print(kod);
      } else if (kod == '5') {
        for (int i = 2; i <= 5; i++) digitalWrite(i, HIGH);
        digitalWrite(10, HIGH);
        for (int i = 6; i <= 9; i++) digitalWrite(i, LOW);
        for (int i = 11; i <= 13; i++) digitalWrite(i, LOW);
        lcd.setCursor(0,1);
        lcd.print("Felkapcsolva :");
        lcd.print(kod);
      } else if (kod == '6') {
        for (int i = 2; i <= 5; i++) digitalWrite(i, HIGH);
        digitalWrite(10, HIGH);
        digitalWrite(13, HIGH);
        digitalWrite(11, LOW);
        digitalWrite(12, LOW);
        lcd.setCursor(0,1);
        lcd.print("Felkapcsolva :");
        lcd.print(kod);
      } else if (kod == '7') {
        for (int i = 2; i <= 5; i++) digitalWrite(i, HIGH);
        digitalWrite(10, HIGH);
        digitalWrite(13, HIGH);
        digitalWrite(11, HIGH);
        digitalWrite(12, LOW);
        lcd.setCursor(0,1);
        lcd.print("Felkapcsolva :");
        lcd.print(kod);
      } else if (kod == '8') {
        for (int i = 2; i <= 5; i++) digitalWrite(i, HIGH);
        digitalWrite(10, HIGH);
        digitalWrite(13, HIGH);
        digitalWrite(11, HIGH);
        digitalWrite(12, HIGH);
        lcd.setCursor(0,1);
        lcd.print("Felkapcsolva :");
        lcd.print(kod);        
      } else if (kod == '0') {
        for (int i = 2; i <= 5; i++) digitalWrite(i, LOW);
        for (int i = 10; i <= 13; i++) digitalWrite(i, LOW);
        lcd.setCursor(0,1);
        lcd.print("Osszes lekapcsolva"); 
      }
    }
  }
}
